193 /**
194 * Читает следующую строку из файла и разбивает её на отдельные значения.
195 * @return true, если строка успешно прочитана, иначе false.
196 */
197 bool ReadNextRow() {
198 if (!std::getline(file_, current_line_)) {
199 isEOF_ = true;
200 return false;
201 }
202
203 std::stringstream ss(current_line_);
204 std::string temp_value;
205 current_values_.clear();
206
207 while (std::getline(ss, temp_value, column_separator_)) {
208 for (size_t i = 0; i < temp_value.size(); ++i) {
209 if (temp_value[i] == escape_character_) {
210 inside_escape_character_ = !inside_escape_character_;
211 temp_value.erase(i, 1);
212 --i;
213 }
214 }
215
216 if (inside_escape_character_) {
217 current_values_.back() += column_separator_ + temp_value;
218 } else {
219 current_values_.push_back(temp_value);
220 }
221 }
222 ToZeroColumn();
223 IncRow();
224 return true;
225 }

254 IncColumn();

257 return value;

266 } else {
267 throw std::runtime_error("Unsupported type");
268 }
269 } catch (const std::exception &ex) {

271 ex.what() << std::endl << kRESET;
272 }
273 }

289 row_separator_ << ", " << "escape character: " << escape_character_ << kRESET << std::endl;
290 }
